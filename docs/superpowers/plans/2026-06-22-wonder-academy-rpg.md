# Wonder Academy RPG Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Replace the current Monster Academy prototype with a Wonder Academy vertical slice that supports starter choice, node-based region movement, a first Mood Trial, Firestore-backed progress, pause/resume, current objectives, and cozy audio.

**Architecture:** Build Wonder Academy as a new self-contained `src/components/LittleGames/wonder-academy/` module. Pure TypeScript data, reducers, save parsing, and audio settings are tested with Vitest; React owns the standalone shell, auth/save status, panels, and overlays; KAPLAY owns the title, hub, region map, and Mood Trial canvas. Firestore is the durable save source, with localStorage only as startup cache and offline pending queue.

**Tech Stack:** React 19, React Router v7, TypeScript strict, Vite, Vitest, Firebase Auth + Firestore, KAPLAY, Tailwind v4 + DaisyUI, lucide-react, Web Audio API.

---

## Scope

This plan implements the first playable vertical slice, not the entire 100-130 Wonderling game. It must not shrink the world design: data types and file boundaries should support future chapters, careers, Mythlings, Guest Wonderling Slots, and expansion content. The shipped slice covers:

- `/games/wonder-academy` standalone route.
- Redirect from `/games/monster-academy` to `/games/wonder-academy`.
- Starter selection with nickname for Lumi, Momo, Pico, Nibi.
- Academy Hub shell with Continue/current objective/save status.
- Sparkleaf Grove node map with adjacent-node movement.
- One playable Mood Trial and Attune outcome.
- Firestore save/load with local cache and pending offline queue.
- Pause menu and conservative audio settings.
- Generated placeholder audio assets from a local script.
- Original visual assets generated by Codex image generation, with stable filenames.

## Global Constraints

- Follow `AGENTS.md`: 2-space indentation, TypeScript strict, functional components, Tailwind utilities, `logger` for logging instead of raw app logging.
- Existing web app has no test runner today. This plan adds Vitest because the spec explicitly requires unit coverage for save parsing, progression, current objectives, audio settings, and Mood Trial logic.
- Do not use third-party IP names/assets. Do not keep Monster Academy wording in user-facing Wonder Academy UI.
- Do not make localStorage the durable save. It is only a cache/pending queue.
- Do not require OpenAI API keys or third-party audio API keys. Audio placeholders are generated by a local Node script.
- Use `npm run test`, `npm run build`, and `npm run lint` before each commit.
- Use Conventional Commits.

export function getCurrentObjective(pointer: WonderAcademyProgressPointer): WonderAcademyObjective {
  const chapter = getChapterById(pointer.currentChapterId) ?? WONDER_ACADEMY_CHAPTERS[0];
  const node =
    chapter.nodes.find((item) => item.id === pointer.currentNodeId) ?? chapter.nodes[0];

  if (pointer.completedNodeIds.includes(node.id)) {
    const nextNode = chapter.nodes.find(
      (candidate) =>
        node.adjacentNodeIds.includes(candidate.id) &&
        !pointer.completedNodeIds.includes(candidate.id) &&
        !candidate.lockedBy,
    );
    return (nextNode ?? node).objective;
  }

  return node.objective;
}

function readStoredProgress(storage: Storage | null, key: string): WonderAcademyProgress | null {
  if (!storage) return null;

  if (!raw) return null;

  } catch {

function isNewer(left: WonderAcademyProgress | null, right: WonderAcademyProgress | null): boolean {
  if (!left) return false;
  if (!right) return true;

export function parseWonderAcademyProgress(input: unknown): WonderAcademyProgress | null {
  if (!input || typeof input !== "object") return null;

  if (
    value.schemaVersion !== 1 ||
    typeof value.userId !== "string" ||
    typeof value.createdAt !== "string" ||
    typeof value.updatedAt !== "string" ||
    typeof value.starterSpeciesId !== "string" ||
    typeof value.starterNickname !== "string" ||
    !value.storyProgress ||

    !Array.isArray(value.completedNodeIds) ||

    return null;

  return value as WonderAcademyProgress;
}

export async function getFirestoreWonderAcademyProgress(
  userId: string,
): Promise<WonderAcademyProgress | null> {
  const docRef = doc(db, "gameProgress", userId, "littleGames", "wonderAcademy");
  const snapshot = await getDoc(docRef);
  return snapshot.exists() ? parseWonderAcademyProgress(snapshot.data()) : null;

export async function setFirestoreWonderAcademyProgress(
  userId: string,
  progress: WonderAcademyProgress,
): Promise<void> {
  const docRef = doc(db, "gameProgress", userId, "littleGames", "wonderAcademy");
  await setDoc(
    docRef,
    {
      ...progress,
      lastCloudSavedAt: progress.updatedAt,
      serverUpdatedAt: serverTimestamp(),
    },
    { merge: true },
  );
}

export function createWonderAcademyProgressService(deps: SaveDeps) {
  return {
    async load(userId: string): Promise<LoadResult> {
      const [cloud, pending, cache] = await Promise.all([
        deps.getCloudProgress(userId),
        Promise.resolve(readStoredProgress(deps.storage, WONDER_ACADEMY_PENDING_KEY)),
        Promise.resolve(readStoredProgress(deps.storage, WONDER_ACADEMY_CACHE_KEY)),
      ]);

      if (isNewer(pending, cloud)) {
        return { progress: pending, source: "pending-local" };
      }
      if (cloud) {
        writeStoredProgress(deps.storage, WONDER_ACADEMY_CACHE_KEY, cloud);
        return { progress: cloud, source: "cloud" };
      }
      if (cache) return { progress: cache, source: "cache" };
      return { progress: null, source: "empty" };
    },

    async save(userId: string, progress: WonderAcademyProgress): Promise<SaveResult> {
      const nextProgress = {
        ...progress,
        userId,
        updatedAt: deps.now(),
      };
      writeStoredProgress(deps.storage, WONDER_ACADEMY_CACHE_KEY, nextProgress);

      try {
        await deps.setCloudProgress(userId, nextProgress);
        deps.storage?.removeItem(WONDER_ACADEMY_PENDING_KEY);
        return { status: "saved", pending: false };
      } catch {
        writeStoredProgress(deps.storage, WONDER_ACADEMY_PENDING_KEY, nextProgress);
        return { status: "pending", pending: true };
      }
    },
  };
}

export const wonderAcademyProgressService = createWonderAcademyProgressService({
  storage: getWonderAcademyStorage(),
  getCloudProgress: getFirestoreWonderAcademyProgress,
  setCloudProgress: setFirestoreWonderAcademyProgress,
  now: () => new Date().toISOString(),
});

  if (value <= 0) return "open";
  if (value <= 2) return "calm";
  if (value <= 4) return "curious";
  if (value <= 6) return "guarded";
  return "upset";

function updateMoodTrial(state: WonderAcademyState, moodShift: number, message: string): WonderAcademyState {
  if (!state.moodTrial) return state;
  const opponentMood = Math.max(0, state.moodTrial.opponentMood - moodShift);
  const opponentMoodState = moodStateFor(opponentMood);
  return {

    message,

      ...state.moodTrial,
      opponentMood,
      opponentMoodState,

export function selectAdjacentNodeIds(state: WonderAcademyState): string[] {
  const node = getNodeById(state.currentChapterId, state.currentNodeId);
  if (!node) return [];
  return node.adjacentNodeIds.filter((nodeId) => state.unlockedNodeIds.includes(nodeId));

export function applyWonderAcademyAction(
  state: WonderAcademyState,
  action: WonderAcademyAction,
): WonderAcademyState {
  if (action.type === "togglePause") {
    return { ...state, paused: !state.paused };
  }

  if (state.paused && action.type !== "togglePause") {
    return state;
  }

  switch (action.type) {
    case "newGame":
      return { ...state, mode: "starterSelect", message: "選擇你的第一位 Wonderling 夥伴。" };
    case "continue":
      return { ...state, mode: "hub", message: "歡迎回來。下一步已經幫你標好了。" };
    case "openRegionMap":
      return { ...state, mode: "regionMap", message: state.currentObjective.description };
    case "returnHub":
      return { ...state, mode: "hub", moodTrial: null, message: "回到 Academy Hub。進度已保存。" };
    case "moveToNode": {
      if (!selectAdjacentNodeIds(state).includes(action.nodeId)) {
        return { ...state, message: "還不能前往這個節點。先跟著目前目標走吧。" };
      }
      const next = {
        ...state,
        mode: "regionMap" as const,
        currentNodeId: action.nodeId,
        message: getNodeById(state.currentChapterId, action.nodeId)?.objective.description ?? state.message,
      };
      return { ...next, currentObjective: buildObjective(next) };
    }
    case "startMoodTrial":
      return {
        ...state,
        mode: "moodTrial",
        currentNodeId: action.nodeId,
        message: "Mossmew 看起來有點緊張。先用 Comfort 安撫牠。",
        moodTrial: {
          nodeId: action.nodeId,
          opponentSpeciesId: "mossmew",
          opponentMood: 5,
          opponentMoodState: "guarded",
          teamMood: 10,
          canAttune: false,
        },
      };
    case "comfort":
      return updateMoodTrial(state, 2, "你的聲音讓 Mossmew 放鬆了一點。");
    case "skill":
      return updateMoodTrial(state, action.skillId === "tiny-flash" ? 3 : 2, "Wonderling 的技能閃出溫柔的光。");
    case "snack":
      return updateMoodTrial(state, action.snackId ? 1 : 0, "Snack 讓氣氛變得更安心。");
    case "attune": {
      if (!state.moodTrial?.canAttune) {
        return { ...state, message: "還沒有到可以 Attune 的時候。" };
      }
      const nextCompletedNodeIds = Array.from(new Set([...state.completedNodeIds, state.moodTrial.nodeId]));
      const nextUnlockedNodeIds = Array.from(new Set([...state.unlockedNodeIds, "mossy-bridge"]));
      const next = {
        ...state,
        mode: "regionMap",
        moodTrial: null,
        completedNodeIds: nextCompletedNodeIds,
        unlockedNodeIds: nextUnlockedNodeIds,
        wonderdex: {
          ...state.wonderdex,
          mossmew: "attuned",
        },
        message: "Attune 成功。Mossmew 願意加入 Wonderdex 了。",
      };
      return { ...next, currentObjective: buildObjective(next) };
    }
    default:
      return state;
  }
}

import { mkdirSync, writeFileSync } from "node:fs";
import { dirname, join } from "node:path";
import { fileURLToPath } from "node:url";

const root = dirname(fileURLToPath(import.meta.url));
const outDir = join(root, "..", "src", "assets", "games", "wonder-academy", "audio");
const sampleRate = 44100;

function writeWav(name, seconds, notes) {
  const sampleCount = Math.floor(sampleRate * seconds);
  const data = Buffer.alloc(sampleCount * 2);

  for (let i = 0; i < sampleCount; i += 1) {
    const t = i / sampleRate;
    const note = notes[Math.min(notes.length - 1, Math.floor((t / seconds) * notes.length))];
    const env = Math.min(1, t * 20) * Math.min(1, (seconds - t) * 8);
    const value =
      Math.sin(2 * Math.PI * note * t) * 0.24 * env +
      Math.sin(2 * Math.PI * note * 2 * t) * 0.06 * env;
    data.writeInt16LE(Math.max(-1, Math.min(1, value)) * 32767, i * 2);
  }

  const header = Buffer.alloc(44);
  header.write("RIFF", 0);
  header.writeUInt32LE(36 + data.length, 4);
  header.write("WAVE", 8);
  header.write("fmt ", 12);
  header.writeUInt32LE(16, 16);
  header.writeUInt16LE(1, 20);
  header.writeUInt16LE(1, 22);
  header.writeUInt32LE(sampleRate, 24);
  header.writeUInt32LE(sampleRate * 2, 28);
  header.writeUInt16LE(2, 32);
  header.writeUInt16LE(16, 34);

  warden_trial_loop: new URL("../../../assets/games/wonder-academy/audio/warden_trial_loop.wav", import.meta.url).href,

function clampVolume(value: unknown, fallback: number): number {
  return typeof value === "number" && Number.isFinite(value)
    ? Math.min(1, Math.max(0, value))
    : fallback;
}

export function normalizeAudioSettings(
  settings: Partial<WonderAcademyAudioSettings> | null,
): WonderAcademyAudioSettings {
  return {
    musicVolume: clampVolume(settings?.musicVolume, DEFAULT_SETTINGS.musicVolume),
    sfxVolume: clampVolume(settings?.sfxVolume, DEFAULT_SETTINGS.sfxVolume),
    muted: settings?.muted ?? DEFAULT_SETTINGS.muted,
  };
}

export function createWonderAcademyAudio(initialSettings: Partial<WonderAcademyAudioSettings> | null) {
  let settings = normalizeAudioSettings(initialSettings);
  let audioContext: AudioContext | null = null;
  let loopAudio: HTMLAudioElement | null = null;

  const ensureContext = () => {
    audioContext ??= new AudioContext();
    if (audioContext.state === "suspended") {
      void audioContext.resume();
    }
  };

  return {
    updateSettings(next: Partial<WonderAcademyAudioSettings>) {
      settings = normalizeAudioSettings({ ...settings, ...next });
      if (loopAudio) {
        loopAudio.volume = settings.muted ? 0 : settings.musicVolume;
      }
      return settings;
    },
    getSettings() {
      return settings;
    },
    playSfx(id: WonderAcademySfxId) {
      if (settings.muted || settings.sfxVolume <= 0) return;
      ensureContext();
      const audio = new Audio(AUDIO_URLS[id]);
      audio.volume = settings.sfxVolume;
      void audio.play();
    },
    playLoop(id: WonderAcademyLoopId) {
      ensureContext();
      loopAudio?.pause();
      loopAudio = new Audio(AUDIO_URLS[id]);
      loopAudio.loop = true;
      loopAudio.volume = settings.muted ? 0 : settings.musicVolume;
      void loopAudio.play();
    },
    stopLoop() {
      loopAudio?.pause();
      loopAudio = null;
    },
  };
}

export function projectWonderAcademyProgress(
